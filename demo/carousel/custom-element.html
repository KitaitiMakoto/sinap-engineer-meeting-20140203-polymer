var bootstrapCarouselCount = 0;
var bootstrapCarouselPrototype = Object.create(HTMLElement.prototype);
bootstrapCarouselPrototype.createdCallback = function() {

  var interval = this.hasAttribute("interval") ? this.getAttribute("interval") : 5000;

  this.innerHTML =
    '<div class="carousel slide" id="bootstrap-carousel-' + bootstrapCarouselCount + '"><ol class="carousel-indicators"><li data-target="#bootstrap-carousel-'+ bootstrapCarouselCount +'" data-slide-to="0" class="active"></li><li data-target="#bootstrap-carousel-' + bootstrapCarouselCount + '" data-slide-to="1"></li><li data-target="#bootstrap-carousel-' + bootstrapCarouselCount + '" data-slide-to="2"></li></ol><div class="carousel-inner">' +
    this.innerHTML +
    '</div><a class="left carousel-control" href="#bootstrap-carousel-' + bootstrapCarouselCount + '" data-slide="prev"><span class="glyphicon glyphicon-chevron-left"></span></a><a class="right carousel-control" href="#bootstrap-carousel-' + bootstrapCarouselCount + '" data-slide="next"><span class="glyphicon glyphicon-chevron-right"></span></a></div>';
  $(this).carousel({interval: interval});
  bootstrapCarouselCount++;
};
var bootstrapCarousel = document.registerElement("bootstrap-carousel", {prototype: bootstrapCarouselPrototype});
